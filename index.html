        function getChatHistory(maxIndex) {

            const formFields = intakeData.currentFormFields;

            const history = [];

            for (let i = 0; i < formFields.length; i++) {

                const field = formFields[i];

                if (i <= maxIndex) {

                    // Use prompt_key here

                    history.push({ speaker: 'ai', prompt_key: field.prompt_key, jargon: field.jargon, id: field.id });

                }

                if (intakeData[field.id] && i < maxIndex) {

                    history.push({ speaker: 'user', text: intakeData[field.id] });

                }

            }

            return history;

        }

                }

            }
